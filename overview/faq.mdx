<Accordion title="What is haunt.gg?">
**haunt.gg** is a biolink platform that allows you to create customizable profiles with features such as **social links, effects, themes, and more**.  
It's designed to help you showcase your online presence effectively.
</Accordion>

<Accordion title="Is haunt.gg free?">
Yes — haunt.gg is completely **free** to use. You can create and customize your profile without any cost or hidden fees.  
We also offer a [**Premium plan**](../guides/premium) with additional features for those who want more customization. **95%** of the platform remains free.
</Accordion>

<Accordion title="Where can I buy Premium or other products?">
You can purchase all products on our [**Pricing page**](https://haunt.gg/pricing).  
You can also **gift products** to other users, which will earn you the [**Gifter badge**](https://help.haunt.gg/customization/badges#gifter-badge).
</Accordion>

<Accordion title="Which payment methods are accepted?">
- **Stripe**  
  - Credit/Debit Card, Google Pay, eps-Transfer (Bank Transfer), Klarna, Apple Pay, Giropay  
- **Cryptocurrencies (via Cryptomus)**  
  - USDT, BTC, LTC, ETH, XMR, POL, BNB, USDC, SOL  
- **PayPal**  
  - Processed through our trusted Point of Sale software, Paddle
</Accordion>

<Accordion title="Is multiaccounting allowed?">
Yes, multiaccounting is allowed with certain limitations. [**Read more**](../guides/multiaccounting).
</Accordion>

<Accordion title="Is imagehost free?">
Yes, our **image host** is free to use.  
You can increase storage limits by purchasing [**Premium**](https://haunt.gg/pricing) or the [**Image Host upgrade**](https://haunt.gg/pricing).  
View all limits [**here**](https://help.haunt.gg/guides/imagehost#limits).
</Accordion>

The **OG badge** is awarded to users who have been on haunt.gg since the beginning.

</Accordion>

Yes. Our API lets you fetch **user** and **leaderboard** data. [**Read more**](../endpoint/introduction.mdx).

<Accordion title="Can I report content on haunt.gg?">
Yes. If you find content that violates our [**Terms of Service**](https://haunt.gg/terms), you can report it:

- Right-click anywhere on a user's profile.  

**Imagehost uploads**  
- Click the flag icon when viewing the image.  

**Assets/Templates**  
- Report via ticket in our [**Discord server**](https://discord.gg/hauntbio).

<script dangerouslySetInnerHTML={{
  __html: `
    (function() {
      let clickedAccordion = null;
      
      // Accordion functionality
      document.addEventListener('click', function(e) {
        const summary = e.target.closest('summary');
        if (summary) {
          clickedAccordion = summary.parentElement;
          
          // Wait for the accordion state to settle
          setTimeout(() => {
            const allAccordions = document.querySelectorAll('details');
            
            allAccordions.forEach(accordion => {
              if (accordion !== clickedAccordion && accordion.hasAttribute('open')) {
                accordion.removeAttribute('open');
              }
            });
            
            clickedAccordion = null;
          }, 100);
        }
      });

      // Search functionality
      document.addEventListener('DOMContentLoaded', function() {
        const searchInput = document.getElementById('faq-search');
        const accordions = document.querySelectorAll('details');
        
        if (!searchInput) return;
        
        // Search through FAQ items
        searchInput.addEventListener('input', function() {
          const searchTerm = this.value.toLowerCase().trim();
          
          accordions.forEach(accordion => {
            const title = accordion.querySelector('summary').textContent.toLowerCase();
            const content = accordion.querySelector('div').textContent.toLowerCase();
            
            if (searchTerm === '' || title.includes(searchTerm) || content.includes(searchTerm)) {
              accordion.style.display = 'block';
              
              // Highlight matching text if search term exists
              if (searchTerm !== '') {
                highlightText(accordion, searchTerm);
              } else {
                removeHighlights(accordion);
              }
            } else {
              accordion.style.display = 'none';
              accordion.removeAttribute('open');
            }
          });
          
          // Show "No results" message if no accordions are visible
          updateNoResultsMessage(searchTerm);
        });
        
        function highlightText(accordion, searchTerm) {
          const summary = accordion.querySelector('summary');
          const content = accordion.querySelector('div');
          
          [summary, content].forEach(element => {
            const walker = document.createTreeWalker(
              element,
              NodeFilter.SHOW_TEXT,
              null,
              false
            );
            
            const textNodes = [];
            let node;
            while (node = walker.nextNode()) {
              textNodes.push(node);
            }
            
            textNodes.forEach(textNode => {
              const text = textNode.textContent;
              const regex = new RegExp(
  '(' + searchTerm.replace(/[\.\*\+\?\^\$\{\}\(\)\|\[\]\\]/g, '\\$&') + ')',
  'gi'
);

              
              if (regex.test(text)) {
                const highlightedText = text.replace(regex, '<mark class="faq-search-highlight">$1</mark>');
                const span = document.createElement('span');
                span.innerHTML = highlightedText;
                textNode.parentNode.replaceChild(span, textNode);
              }
            });
          });
        }
        
        function removeHighlights(accordion) {
          const marks = accordion.querySelectorAll('mark');
          marks.forEach(mark => {
            const parent = mark.parentNode;
            parent.replaceChild(document.createTextNode(mark.textContent), mark);
            parent.normalize();
          });
        }
        
        function updateNoResultsMessage(searchTerm) {
          let noResultsMsg = document.getElementById('no-results-message');
          const visibleAccordions = Array.from(accordions).filter(acc => acc.style.display !== 'none');
          
          if (searchTerm !== '' && visibleAccordions.length === 0) {
            if (!noResultsMsg) {
              noResultsMsg = document.createElement('div');
              noResultsMsg.id = 'no-results-message';
              noResultsMsg.innerHTML = '🔍 No FAQ items found matching your search.';
              document.querySelector('AccordionGroup').parentNode.appendChild(noResultsMsg);
            }
            noResultsMsg.style.display = 'block';
          } else if (noResultsMsg) {
            noResultsMsg.style.display = 'none';
          }
        }
      });
    })();
  `
}} />
